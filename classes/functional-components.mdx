```js title="useState"
function Contador() {
  const [count, setCount] = useState(0);

  const incrementa = () => {
    count++;
  };

  return (
    <div>

      </button>
    </div>
  );
```

```js title="useState"
import { useState } from 'react';

function Contador() {
  const [count, setCount] = useState(0);

  const incrementa = () => {
    setCount(count + 1);
  };

  return (
    <div>
      <span>Count: {count}</span><br/>
      <button onClick={incrementa}>
        Incrementa
      </button>
    </div>
  );
}
```
